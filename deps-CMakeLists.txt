cmake_minimum_required(VERSION 3.2)
project(deps-download LANGUAGES NONE)

include(ExternalProject)
set_directory_properties(PROPERTIES EP_BASE ".")

ExternalProject_Add (
    GLFW_EX
    GIT_REPOSITORY "https://github.com/glfw/glfw.git"
    GIT_TAG "master"
    CMAKE_ARGS -DGLFW_BUILD_EXAMPLES=OFF
        -DGLFW_BUILD_TESTS=OFF
        -DGLFW_BUILD_DOCS=OFF
        -DGLFW_INSTALL=OFF
        -DBUILD_SHARED_LIBS=OFF
	BUILD_COMMAND		${CMAKE_COMMAND} --build . --config Debug
						COMMAND ${CMAKE_COMMAND} --build . --config Release
    UPDATE_COMMAND    ""
    INSTALL_COMMAND   ""
    TEST_COMMAND      "")

ExternalProject_Add (
       GLM_EX
       GIT_REPOSITORY "https://github.com/g-truc/glm.git"
       GIT_TAG "master"
       UPDATE_COMMAND    ""
       BUILD_COMMAND     ""
       INSTALL_COMMAND   ""
       TEST_COMMAND      "")

ExternalProject_Add(
    GLEW_EX
    GIT_REPOSITORY "https://github.com/Perlmint/glew-cmake.git"
    GIT_TAG "master"
    CMAKE_ARGS	-Dglew-cmake_BUILD_SINGLE_CONTEXT=ON
                -Dglew-cmake_BUILD_STATIC=ON
	BUILD_COMMAND		${CMAKE_COMMAND} --build . --config Debug
						COMMAND ${CMAKE_COMMAND} --build . --config Release
    UPDATE_COMMAND		""
    INSTALL_COMMAND   	""
    TEST_COMMAND      	"")
	
if(NOT EXISTS "Source/STB_IMAGE_EX/stb_image.h")
	file(DOWNLOAD
		"https://raw.githubusercontent.com/nothings/stb/master/stb_image.h"
		"Source/STB_IMAGE_EX/stb_image.h"
		SHOW_PROGRESS)
endif()
